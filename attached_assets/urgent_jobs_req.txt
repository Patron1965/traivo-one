TRAIVO GO – Akut Jobbhantering för
Fältpersonal (Kravdokument för Replit)
System: Traivo Go (mobilapp för fältpersonal/tekniker/chaufförer)
Prioritet: 🔴 Hög
Datum: 2026-03-27
Mål: Implementera mottagning, accept och hantering av akuta jobb tilldelade från Traivo One-planer‐
aren

1. Användarscenario
Vem är användaren?
Tekniker/chaufför i fält som kör mellan uppdrag. Hen har Traivo Go-appen öppen på en monterad tele‐
fon i fordonet. Under dagen kan planeraren tilldela akuta jobb som måste hanteras snabbt.

Typisk situation
• Teknikern kör mot sitt nästa planerade uppdrag
• Planeraren ser att ett akut jobb har inkommit 1.2 km bort
• Planeraren tilldelar jobbet till teknikern via Traivo One
• Teknikern får en tydlig push-notis och måste snabbt kunna acceptera eller avböja
• Vid accept: navigering startas, planerad rutt uppdateras automatiskt

Huvudproblem att lösa
• Teknikern måste kunna se och reagera på akut jobb medan hen kör (hands-free/minimal interak‐
tion)
• Accept/decline måste gå på maximalt 2 tryck
• Jobbet måste prioriteras om i uppdragslistan automatiskt
• Offline-stöd: Om teknikern tillfälligt tappar uppkoppling ska jobbet synkas när nätet kommer till‐
baka

3.2 WebSocket – Realtidskommunikation
// hooks/useUrgentJobSocket.ts
import { useEffect, useCallback } from 'react';
import { useSocket } from './useSocket';
import { useUrgentJobStore } from '../stores/urgentJobStore';
export function useUrgentJobSocket() {
const socket = useSocket();
const { setIncomingJob, clearIncomingJob } = useUrgentJobStore();
useEffect(() => {
// Lyssna på tilldelade akuta jobb
socket.on('job:urgent:assigned', (data: UrgentJobAssignment) => {
setIncomingJob(data);
showUrgentJobNotification(data.job);
playUrgentSound();
triggerHapticFeedback('heavy');
});
// Bekräftelse att accept/decline mottogs
socket.on('job:urgent:response:ack', (data: { jobId: string; status: string }) =>
{
console.log(`Akut jobb ${data.jobId} response bekräftad: ${data.status}`);
});
return () => {
socket.off('job:urgent:assigned');
socket.off('job:urgent:response:ack');
};
}, [socket]);
const acceptJob = useCallback((jobId: string, startNavigation: boolean) => {
socket.emit('job:urgent:accept', { jobId, startNavigation });
clearIncomingJob();
}, [socket]);
const declineJob = useCallback((jobId: string, reason: string) => {
socket.emit('job:urgent:decline', { jobId, reason });
clearIncomingJob();
}, [socket]);
return { acceptJob, declineJob };
}

3.3 Realtidspositionering (GPS)
// hooks/useGpsReporting.ts
import * as Location from 'expo-location';
export function useGpsReporting() {
useEffect(() => {
let subscription: Location.LocationSubscription;
const startTracking = async () => {
const { status } = await Location.requestForegroundPermissionsAsync();
if (status !== 'granted') return;
// Bakgrundsposition för kontinuerlig rapportering
await Location.requestBackgroundPermissionsAsync();
subscription = await Location.watchPositionAsync(
{
accuracy: Location.Accuracy.High,
distanceInterval: 50, // Minst 50m mellan uppdateringar
timeInterval: 10000, // Max var 10:e sekund
},
(location) => {
socket.emit('technician:position', {
lat: location.coords.latitude,
lng: location.coords.longitude,
speed: location.coords.speed,
heading: location.coords.heading,
timestamp: location.timestamp,
});
}
);
};
startTracking();
return () => subscription?.remove();
}, []);
}

// services/offlineQueue.ts
import AsyncStorage from '@react-native-async-storage/async-storage';
import NetInfo from '@react-native-community/netinfo';
interface QueuedAction {
id: string;
type: 'accept' | 'decline' | 'status_update';
payload: any;
timestamp: number;
retries: number;
}
class OfflineQueue {
private queue: QueuedAction[] = [];
private isOnline = true;
constructor() {
// Lyssna på nätverksstatus
NetInfo.addEventListener(state => {
const wasOffline = !this.isOnline;
this.isOnline = state.isConnected ?? false;
// Återfick uppkoppling → synka kö
if (wasOffline && this.isOnline) {
this.processQueue();
}
});
// Ladda sparad kö vid start
this.loadQueue();
}
async enqueue(action: Omit<QueuedAction, 'id' | 'timestamp' | 'retries'>) {
const item: QueuedAction = {
...action,
id: generateId(),
timestamp: Date.now(),
retries: 0,
};
this.queue.push(item);
await this.saveQueue();
if (this.isOnline) {
this.processQueue();
}
}
private async processQueue() {
while (this.queue.length > 0) {
const item = this.queue[0];
try {
await this.sendToServer(item);
this.queue.shift();
await this.saveQueue();
} catch (error) {
item.retries++;
if (item.retries > 5) {
this.queue.shift(); // Ge upp efter 5 försök
logError('Offline queue item failed permanently', item);
}
break; // Försök igen senare
}

10
}
}
private async sendToServer(item: QueuedAction) {
// Skicka via WebSocket eller REST API
switch (item.type) {
case 'accept':
await api.post('/jobs/urgent/accept', item.payload);
break;
case 'decline':
await api.post('/jobs/urgent/decline', item.payload);
break;
case 'status_update':
await api.post('/jobs/status', item.payload);
break;
}
}
private async saveQueue() {
await AsyncStorage.setItem('offline_queue', JSON.stringify(this.queue));
}
private async loadQueue() {
const data = await AsyncStorage.getItem('offline_queue');
if (data) this.queue = JSON.parse(data);
}
}
export const offlineQueue = new OfflineQueue();

3.5 Geofence – Automatisk “Framme”-detektion
// services/geofence.ts
import * as Location from 'expo-location';
import * as TaskManager from 'expo-task-manager';
const GEOFENCE_TASK = 'ARRIVAL_GEOFENCE';
const ARRIVAL_RADIUS = 50; // meter
TaskManager.defineTask(GEOFENCE_TASK, ({ data, error }) => {
if (error) return;
const { eventType, region } = data as any;
if (eventType === Location.GeofencingEventType.Enter) {
// Teknikern har anlänt till jobbplatsen
updateJobStatus(region.identifier, 'arrived');
showLocalNotification('📍 Du har anlänt till jobbplatsen');
}
});
export async function setArrivalGeofence(jobId: string, lat: number, lng: number) {
await Location.startGeofencingAsync(GEOFENCE_TASK, [
{
identifier: jobId,
latitude: lat,
longitude: lng,
radius: ARRIVAL_RADIUS,
notifyOnEnter: true,
},
]);
}
